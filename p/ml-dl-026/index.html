def gini_impurity(y):
    '''ËÆ°ÁÆóÂü∫Â∞ºÁ≥ªÊï∞'''
    m = len(y)
    if m == 0:
        return 0
    p = np.bincount(y) / m
    return 1 - np.sum(p**2)

def split_dataset(X, y, feature, threshold):
    '''Ê†πÊçÆÁâπÂæÅÂíåÈòàÂÄºÂàÜÂâ≤Êï∞ÊçÆÈõÜ'''
    left_mask = X[:, feature] <= threshold
    right_mask = X[:, feature] > threshold
    return X[left_mask], y[left_mask], X[right_mask], y[right_mask]
