import requests
from bs4 import BeautifulSoup
import pandas as pd
import time
import matplotlib.pyplot as plt
import seaborn as sns

class DoubanMovieSpider:
    def __init__(self):
        self.base_url = 'https://movie.douban.com/top250'
        self.headers = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        }
        self.movies = []
    
    def get_page(self, start):
        '''Ëé∑ÂèñÂçïÈ°µÊï∞ÊçÆ'''
        url = f'{self.base_url}?start={start}'
        try:
            response = requests.get(url, headers=self.headers)
            response.raise_for_status()
            return response.text
        except Exception as e:
            print(f'ËØ∑Ê±ÇÂ§±Ë¥•: {e}')
            return None
    
    def parse_page(self, html):
        '''Ëß£ÊûêÈ°µÈù¢Êï∞ÊçÆ'''
        soup = BeautifulSoup(html, 'html.parser')
        items = soup.find_all('div', class_='item')
        
        for item in items:
            movie = {}
            
            # Ê†áÈ¢ò
            title = item.find('span', class_='title').text
            movie['title'] = title
            
            # ËØÑÂàÜ
            rating = item.find('span', class_='rating_num').text
            movie['rating'] = float(rating)
            
            # ËØÑ‰ª∑‰∫∫Êï∞
            people = item.find('div', class_='star').find_all('span')[-1].text
            movie['people'] = people.strip('‰∫∫ËØÑ‰ª∑')
            
            # ÁÆÄ‰ªã
            inq = item.find('span', class_='inq')
            movie['quote'] = inq.text if inq else ''
            
            # ËØ¶ÊÉÖ
            info = item.find('div', class_='bd').find('p').text
            parts = info.split('\n')
            movie['info'] = parts[1].strip() if len(parts) > 1 else ''
            
            self.movies.append(movie)
    
    def crawl(self):
        '''ÂºÄÂßãÁà¨Âèñ'''
        print('ÂºÄÂßãÁà¨ÂèñË±ÜÁì£ÁîµÂΩ±Top250...')
        
        for i in range(0, 250, 25):
            print(f'Ê≠£Âú®Áà¨ÂèñÁ¨¨{i//25 + 1}È°µ...')
            html = self.get_page(i)
            
            if html:
                self.parse_page(html)
                time.sleep(2)  # Á§ºË≤åÂª∂Êó∂
        
        print(f'Áà¨ÂèñÂÆåÊàêÔºÅÂÖ±Ëé∑Âèñ{len(self.movies)}ÈÉ®ÁîµÂΩ±')
    
    def save_to_csv(self, filename='douban_top250.csv'):
        '''‰øùÂ≠ò‰∏∫CSV'''
        df = pd.DataFrame(self.movies)
        df.to_csv(filename, index=False, encoding='utf-8-sig')
        print(f'Êï∞ÊçÆÂ∑≤‰øùÂ≠òÂà∞{filename}')
    
    def analyze(self):
        '''Êï∞ÊçÆÂàÜÊûê'''
        df = pd.DataFrame(self.movies)
        
        # Âü∫Êú¨ÁªüËÆ°
        print('\nÊï∞ÊçÆÊ¶ÇËßàÔºö')
        print(df.describe())
        
        # ËØÑÂàÜÂàÜÂ∏É
        plt.figure(figsize=(12, 5))
        
        plt.subplot(1, 2, 1)
        plt.hist(df['rating'], bins=20, edgecolor='black')
        plt.title('ËØÑÂàÜÂàÜÂ∏É')
        plt.xlabel('ËØÑÂàÜ')
        plt.ylabel('ÁîµÂΩ±Êï∞Èáè')
        
        # Top10ÁîµÂΩ±
        plt.subplot(1, 2, 2)
        top10 = df.nlargest(10, 'rating')
        plt.barh(range(10), top10['rating'])
        plt.yticks(range(10), top10['title'])
        plt.xlabel('ËØÑÂàÜ')
        plt.title('Top10ÁîµÂΩ±')
        plt.gca().invert_yaxis()
        
        plt.tight_layout()
        plt.savefig('douban_analysis.png', dpi=300, bbox_inches='tight')
        plt.show()

# ‰ΩøÁî®Á§∫‰æã
