def create_responsive_layout(n_plots):
    '''Ê†πÊçÆÂõæË°®Êï∞ÈáèËá™Âä®ËÆ°ÁÆóÂ∏ÉÂ±Ä'''
    if n_plots <= 2:
        return 1, n_plots
    elif n_plots <= 4:
        return 2, 2
    elif n_plots <= 6:
        return 2, 3
    else:
        ncols = 3
        nrows = (n_plots + ncols - 1) // ncols
        return nrows, ncols

n = 7
nrows, ncols = create_responsive_layout(n)
fig, axes = plt.subplots(nrows, ncols, figsize=(5*ncols, 4*nrows))
